    nano 2b.sh
    chmod +x 2b.sh

Continut script:

    #! /bin/bash
    cat users.csv | tail -n + 2 > fisier_temp.txt
    while read -r linie;
    do
        utilizator=$(echo $linie | awk -F "," '{print $1}')
        parola=$(echo $linie | awk "," '{print $2}')
        sudo useradd -p $(openssl passwd -1 $parola) $utilizator
    done < fisier_temp.txt
    rm -rf fisier_temp.txt
